[docs]class CRC16(AbstractChecksum):
    r"""This class implements the CRC16 function.

    The constructor expects some parameters:

    :param targets: The targeted fields of the relationship.
    :type targets: a :class:`list` of :class:`Field <netzob.Model.Vocabulary.Field.Field>`, required


    The following example shows how to create a checksum relationship
    with another field:

    >>> from netzob.all import *
    >>> import binascii
    >>> f1 = Field(Raw(b'\xaa\xbb'))
    >>> f2 = Field(CRC16([f1]))
    >>> f = Field([f1, f2])
    >>> binascii.hexlify(next(f.specialize()))
    b'aabb3ed3'

    The following example shows how to create a checksum relationship
    with a group of fields:

    >>> from netzob.all import *
    >>> import binascii
    >>> f1 = Field(Raw(b'\xaa\xbb'))
    >>> f2 = Field(Raw(b'\xcc\xdd'))
    >>> f3 = Field(Raw(b'\xee\xff'))
    >>> f4 = Field(CRC16([f1, f2, f3]))
    >>> f = Field([f1, f2, f3, f4])
    >>> binascii.hexlify(next(f.specialize()))
    b'aabbccddeeff5e9b'

    """

    def calculate(self, msg):
        return _CRC16().calculate(msg)

    def getBitSize(self):
        return UnitSize.SIZE_16.value
